# =============================================================================
# ufc_core - Unified Finite Element Computation Kernel
# Optional: UFC_BUILD_MINIMAL_KERNEL=ON excludes optional modules (see kernel_optional_sources.txt)
# =============================================================================
cmake_minimum_required(VERSION 3.16)

project(ufc_core LANGUAGES Fortran)

set(CMAKE_Fortran_STANDARD 2003)
if(MSVC)
  set(CMAKE_Fortran_FLAGS "${CMAKE_Fortran_FLAGS} /fpp")
else()
  set(CMAKE_Fortran_FLAGS "${CMAKE_Fortran_FLAGS} -cpp")
endif()

# Collect all .f90 under ufc_core (recursive)
file(GLOB_RECURSE UFC_CORE_SRCS "${CMAKE_CURRENT_SOURCE_DIR}/*.f90")

list(FILTER UFC_CORE_SRCS EXCLUDE REGEX ".*/skeleton/.*\\.f90$")

list(FILTER UFC_CORE_SRCS EXCLUDE REGEX ".*/L5_RT/Element/RT_Elem_Acoustic_Stub\\.f90$")

# Exclude L2_NM/ExternalLibs (built as ufc_core_L2_ExternalLibs, migrated from ufc_numcore)

# =============================================================================
# Optional: MPI / CUDA for device capability probes (IF_Base_Domain)
# =============================================================================
option(UFC_HAVE_MPI "Enable MPI probe in IF_Base_Domain (mpiAvailable)" OFF)
option(UFC_HAVE_CUDA "Enable CUDA probe in IF_Base_Domain (gpuAvailable)" OFF)
if(UFC_HAVE_MPI)
  find_package(MPI QUIET COMPONENTS Fortran)
  if(MPI_Fortran_FOUND)
    add_compile_definitions(UFC_HAVE_MPI)
    set(UFC_MPI_LIBS ${MPI_Fortran_LIBRARIES})
    message(STATUS "UFC: MPI probe enabled (IF_Base_Domain)")
  else()
    message(STATUS "UFC: UFC_HAVE_MPI=ON but MPI not found, probe disabled")
  endif()
endif()
if(UFC_HAVE_CUDA)
  add_compile_definitions(UFC_HAVE_CUDA)
  message(STATUS "UFC: CUDA probe placeholder enabled")
endif()

# =============================================================================
# Minimal Kernel Option: Exclude optional modules for embedded/minimal builds
# =============================================================================
# Optional modules:
#   - L1_IF/Optional: Dependency Injection, Serialization, Module Manager, IPC
#   - L5_RT/Optional: Log System, Performance Monitor, Service Registry, IO Restart
#   - L3_MD/Ctx/Keyword: Keyword Registry (preprocessing/frontend only)
#   - L3_MD/Ctx/UI: UI Managers (preprocessing/frontend only)
# See: refactor_out/kernel_optional_sources.txt for complete list
# Note: L3_MD/Ctx files are organized into subdirectories (Base, Model, Part, Mesh, Property, Sets, Contact, Load, Output, Context, Keyword, UI, UniFld). Bridge domain removed 2026-03-09.

  message(STATUS "UFC: ===== MINIMAL KERNEL BUILD =====")
  list(FILTER UFC_CORE_SRCS EXCLUDE REGEX ".*/L1_IF/Optional/.*\\.f90$")
  message(STATUS "UFC:   Excluded L1_IF/Optional/* (Dep_Inj, Serial, Mod_Mgr, IPC)")
  list(FILTER UFC_CORE_SRCS EXCLUDE REGEX ".*/L5_RT/Optional/.*\\.f90$")
  message(STATUS "UFC:   Excluded L5_RT/Optional/* (Log, Performance, Service, IO_Restart)")
  list(FILTER UFC_CORE_SRCS EXCLUDE REGEX ".*/L3_MD/Ctx/Keyword/MD_KW_.*\\.f90$")
  list(FILTER UFC_CORE_SRCS EXCLUDE REGEX ".*/L3_MD/Ctx/UI/MD_UI_.*\\.f90$")
  message(STATUS "UFC:   Excluded L3_MD/Ctx/Keyword/MD_KW_* and L3_MD/Ctx/UI/MD_UI_* (preprocessing modules)")
  message(STATUS "UFC: ===================================")
else()
  message(STATUS "UFC: Full ufc_core build (including optional modules)")
endif()

    target_include_directories(ufc_core_L1_IF PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} "${_CORE_MOD_DIR}")
    set(_L1_OBJS $<TARGET_OBJECTS:ufc_core_L1_IF>)
  endif()

  if(_L2_EXT_SRCS)
    add_library(ufc_core_L2_ExternalLibs OBJECT ${_L2_EXT_SRCS})
    set_target_properties(ufc_core_L2_ExternalLibs PROPERTIES Fortran_MODULE_DIRECTORY "${_CORE_MOD_DIR}")
    target_include_directories(ufc_core_L2_ExternalLibs PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} "${_CORE_MOD_DIR}")
  endif()

  add_library(ufc_core_L2_NM OBJECT ${_L2_SRCS})
  set_target_properties(ufc_core_L2_NM PROPERTIES Fortran_MODULE_DIRECTORY "${_CORE_MOD_DIR}")
  target_include_directories(ufc_core_L2_NM PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} "${_CORE_MOD_DIR}")
  target_link_libraries(ufc_core_L2_NM PUBLIC ufc_core_L1_IF)
  add_dependencies(ufc_core_L2_NM ufc_core_L1_IF)
  if(_L2_EXT_SRCS)
    target_link_libraries(ufc_core_L2_NM PUBLIC ufc_core_L2_ExternalLibs)
    add_dependencies(ufc_core_L2_NM ufc_core_L2_ExternalLibs)
  endif()

  add_library(ufc_core_L3_MD OBJECT ${_L3_SRCS})
  set_target_properties(ufc_core_L3_MD PROPERTIES Fortran_MODULE_DIRECTORY "${_CORE_MOD_DIR}")
  target_include_directories(ufc_core_L3_MD PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} "${_CORE_MOD_DIR}")
  target_link_libraries(ufc_core_L3_MD PUBLIC ufc_core_L1_IF ufc_core_L2_NM)
  add_dependencies(ufc_core_L3_MD ufc_core_L1_IF ufc_core_L2_NM)

  # L4_PH: thin Output + WriteBack only (see UFC_CORE_SRCS re-include after L4_PH blanket exclude)
  if(_L4_SRCS)
    add_library(ufc_core_L4_PH OBJECT ${_L4_SRCS})
    set_target_properties(ufc_core_L4_PH PROPERTIES Fortran_MODULE_DIRECTORY "${_CORE_MOD_DIR}")
    target_include_directories(ufc_core_L4_PH PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} "${_CORE_MOD_DIR}")
    # Thin Output/WriteBack: IF_Prec_Core + IF_Err_Brg only (no L3_MD USE)
    target_link_libraries(ufc_core_L4_PH PUBLIC ufc_core_L1_IF)
    add_dependencies(ufc_core_L4_PH ufc_core_L1_IF)
  endif()

  add_library(ufc_core_L5_RT OBJECT ${_L5_SRCS})
  set_target_properties(ufc_core_L5_RT PROPERTIES Fortran_MODULE_DIRECTORY "${_CORE_MOD_DIR}")
  target_include_directories(ufc_core_L5_RT PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} "${_CORE_MOD_DIR}")
  target_link_libraries(ufc_core_L5_RT PUBLIC ufc_core_L1_IF ufc_core_L2_NM ufc_core_L3_MD)
  add_dependencies(ufc_core_L5_RT ufc_core_L1_IF ufc_core_L2_NM ufc_core_L3_MD)
  if(TARGET ufc_core_L4_PH)
    target_link_libraries(ufc_core_L5_RT PUBLIC ufc_core_L4_PH)
    add_dependencies(ufc_core_L5_RT ufc_core_L4_PH)
  endif()
  if(UFC_SLIM_EXCLUDE_L4_ELEMENT_ACOUSTIC)
    target_compile_definitions(ufc_core_L5_RT PRIVATE UFC_SLIM_NO_ACOUSTIC_ELEM)
  endif()

  add_library(ufc_core_L6_AP OBJECT ${_L6_SRCS})
  set_target_properties(ufc_core_L6_AP PROPERTIES Fortran_MODULE_DIRECTORY "${_CORE_MOD_DIR}")
  target_include_directories(ufc_core_L6_AP PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} "${_CORE_MOD_DIR}")
  target_link_libraries(ufc_core_L6_AP PUBLIC ufc_core_L1_IF ufc_core_L2_NM ufc_core_L3_MD ufc_core_L5_RT)
  add_dependencies(ufc_core_L6_AP ufc_core_L1_IF ufc_core_L2_NM ufc_core_L3_MD ufc_core_L5_RT)
  if(TARGET ufc_core_L4_PH)
    target_link_libraries(ufc_core_L6_AP PUBLIC ufc_core_L4_PH)
    add_dependencies(ufc_core_L6_AP ufc_core_L4_PH)
  endif()

  # L3_MD KeyWord (MD_KW_AP_Brg, MD_KW_Mapper): depends on L6_AP for AP_* modules
  if(_L3_KW_SRCS)
    add_library(ufc_core_L3_KW OBJECT ${_L3_KW_SRCS})
    set_target_properties(ufc_core_L3_KW PROPERTIES Fortran_MODULE_DIRECTORY "${_CORE_MOD_DIR}")
    target_include_directories(ufc_core_L3_KW PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} "${_CORE_MOD_DIR}")
    target_link_libraries(ufc_core_L3_KW PUBLIC ufc_core_L1_IF ufc_core_L2_NM ufc_core_L3_MD ufc_core_L5_RT ufc_core_L6_AP)
    add_dependencies(ufc_core_L3_KW ufc_core_L1_IF ufc_core_L2_NM ufc_core_L3_MD ufc_core_L5_RT ufc_core_L6_AP)
    if(TARGET ufc_core_L4_PH)
      target_link_libraries(ufc_core_L3_KW PUBLIC ufc_core_L4_PH)
      add_dependencies(ufc_core_L3_KW ufc_core_L4_PH)
    endif()
  endif()

  if(_OTHER_SRCS)
    add_library(ufc_core_OTHER OBJECT ${_OTHER_SRCS})
    set_target_properties(ufc_core_OTHER PROPERTIES Fortran_MODULE_DIRECTORY "${_CORE_MOD_DIR}")
    target_include_directories(ufc_core_OTHER PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} "${_CORE_MOD_DIR}")
    target_link_libraries(ufc_core_OTHER PUBLIC ufc_core_L1_IF ufc_core_L2_NM ufc_core_L3_MD ufc_core_L5_RT ufc_core_L6_AP)
    add_dependencies(ufc_core_OTHER ufc_core_L1_IF ufc_core_L2_NM ufc_core_L3_MD ufc_core_L5_RT ufc_core_L6_AP)
    if(TARGET ufc_core_L4_PH)
      target_link_libraries(ufc_core_OTHER PUBLIC ufc_core_L4_PH)
      add_dependencies(ufc_core_OTHER ufc_core_L4_PH)
    endif()
  endif()

  set(_CORE_OBJS
    ${_L1_OBJS}
  )
  if(_L2_EXT_SRCS)
    list(APPEND _CORE_OBJS $<TARGET_OBJECTS:ufc_core_L2_ExternalLibs>)
  endif()
  list(APPEND _CORE_OBJS
    $<TARGET_OBJECTS:ufc_core_L2_NM>
    $<TARGET_OBJECTS:ufc_core_L3_MD>
  )
  if(TARGET ufc_core_L4_PH)
    list(APPEND _CORE_OBJS $<TARGET_OBJECTS:ufc_core_L4_PH>)
  endif()
  list(APPEND _CORE_OBJS
    $<TARGET_OBJECTS:ufc_core_L5_RT>
    $<TARGET_OBJECTS:ufc_core_L6_AP>
  )
  if(_L3_KW_SRCS)
    list(APPEND _CORE_OBJS $<TARGET_OBJECTS:ufc_core_L3_KW>)
  endif()
  if(_CORE_OBJS)
    if(_OTHER_SRCS)
      list(APPEND _CORE_OBJS $<TARGET_OBJECTS:ufc_core_OTHER>)
    endif()
    add_library(ufc_core STATIC ${_CORE_OBJS})
    if(_OTHER_SRCS)
      add_dependencies(ufc_core ufc_core_OTHER)
    else()
      add_dependencies(ufc_core ufc_core_L6_AP)
    endif()
    if(TARGET ufc_core_L4_PH)
      add_dependencies(ufc_core ufc_core_L4_PH)
    endif()
    if(_L3_KW_SRCS)
      add_dependencies(ufc_core ufc_core_L3_KW)
    endif()
  else()
    message(WARNING "UFC: _CORE_OBJS is empty, cannot create ufc_core library")
  endif()
else()
  set(UFC_CORE_SRCS ${_L1_SRCS} ${_L2_EXT_SRCS} ${_L2_SRCS} ${_L3_SRCS} ${_L4_SRCS} ${_L5_SRCS} ${_L6_SRCS} ${_OTHER_SRCS})
  if(UFC_CORE_SRCS)
    add_library(ufc_core STATIC ${UFC_CORE_SRCS})
    set_target_properties(ufc_core PROPERTIES Fortran_MODULE_DIRECTORY "${_CORE_MOD_DIR}")
    if(UFC_SLIM_EXCLUDE_L4_ELEMENT_ACOUSTIC)
      target_compile_definitions(ufc_core PRIVATE UFC_SLIM_NO_ACOUSTIC_ELEM)
    endif()
  else()
    message(WARNING "UFC: No source files found for ufc_core library")
  endif()
endif()
